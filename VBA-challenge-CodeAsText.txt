Sub Ticker():        'Create a script that will loop through all the stocks for one year        'and output the following information:             '(1)The ticker symbol.              '(2)Yearly change from opening price at the beginning of a given                 'year to the closing price at the end of that year.             '(3)The percent change from opening price at the beginning of a                 'given year to the closing price at the end of that year.             '(4)The total stock volume of the stock.     '(5)You should also have conditional formatting that will highlight positive        'change in green and negative change in red. '--------------------------------        '------------------------------    'Had help from a tutor to complete homework    ' --------------------------------------------    ' LOOP THROUGH ALL SHEETS    ' --------------------------------------------    For Each ws In Worksheets        'Column Headings    ws.Range("I1").Value = "Ticker"    ws.Range("J1").Value = "Yearly Change"    ws.Range("K1").Value = "Percent Change"    ws.Range("L1").Value = "Total Stock Volume"    ws.Range("N2").Value = "Greatest % Increase"    ws.Range("N3").Value = "Greatest % Decrease"    ws.Range("N4").Value = "Greatest Total Volume"    ws.Range("O1").Value = "Ticker Symbol"    ws.Range("P1").Value = "Value"                    'Define Ticker as a variable    'https://stackoverflow.com/questions/62471422/vba-loop-how-to-get-ticker-symbols-into-ticker-column    Dim Ticker As String      ' Set an initial variable for holding the total volume per stock  Dim StockVolume As Double  StockVolume = 0    'Define yearly change  Dim YearlyChange As Double  YearlyChange = 0    'Define Percent Change  Dim PercentChange As Double  PercentChange = 0      ' Keep track of the location for each stock in the summary table  Dim Summary_Table_Row As Integer  Summary_Table_Row = 2    'Define Opening value  Dim OpeningValue As Double  OpeningValue = ws.Cells(2, 3).Value    'Define Closing Value  Dim ClosingValue As Double' Determine the Last Row        LastRow = ws.Cells(Rows.Count, 1).End(xlUp).Row   'Loop through ticker column    For i = 2 To LastRow               ' Check if we are still within the same ticker symbol, if it is not...            If ws.Cells(i + 1, 1).Value <> ws.Cells(i, 1).Value Then                                 ' Set the Ticker                    Ticker = ws.Cells(i, 1).Value                      ' Add to the Stock volume                      StockVolume = StockVolume + ws.Cells(i, 7).Value                    ' Print the ticker symbol in the Summary Table                     ws.Range("I" & Summary_Table_Row).Value = Ticker                    ' Print the stock volume to the Summary Table                    ws.Range("L" & Summary_Table_Row).Value = StockVolume                                        'get closing value                    ClosingValue = ws.Cells(i, 6).Value                                        'print the yearly change                    YearlyChange = ClosingValue - OpeningValue                                    ' Print the yearly change in the Summary Table                    ws.Range("J" & Summary_Table_Row).Value = YearlyChange                                        'Conditional formatting for positive change                    If YearlyChange > 0 Then                    ' Color the positive change green                        ws.Range("J" & Summary_Table_Row).Interior.ColorIndex = 4                    ElseIf YearlyChange < 0 Then                'Color the negative change red                        ws.Range("J" & Summary_Table_Row).Interior.ColorIndex = 3                    End If                    'add percentage change                    If OpeningValue = 0 Then                        PercentChange = YearlyChange                                            Else                        PercentChange = YearlyChange / OpeningValue                                            End If                    'Print precent change                    ws.Range("K" & Summary_Table_Row).Value = PercentChange                                         ' Add one to the summary table row                    Summary_Table_Row = Summary_Table_Row + 1                                              ' Reset the stock volume                    StockVolume = 0                                        'reset Opening Value                    OpeningValue = ws.Cells(i + 1, 3).Value        ' If the cell immediately following a row is the same ticker...        Else                ' Add to the Stock Volume                StockVolume = StockVolume + ws.Cells(i, 7).Value        End If          Next i          'Set Percent Format    For i = 2 To LastRow        ws.Cells(i, 11).Style = "Percent"    Next i        '-------------------'find greatest value in a column    'https://stackoverflow.com/questions/42633273/finding-max-of-a-column-in-vba/42633375    ws.Range("P2").Value = Application.WorksheetFunction.Max(ws.Range("k2:k" & Summary_Table_Row))          'Dim GreatestIncrease As Integer     GreatestIncrease = WorksheetFunction.Match(WorksheetFunction.Max(ws.Range("k2:k" & Summary_Table_Row)), ws.Range("k2:k" & Summary_Table_Row), 0)          'print greatest increase ticker symbol     ws.Range("O2").Value = ws.Cells(GreatestIncrease + 1, 9).Value    'find smallest value in a column    'https://stackoverflow.com/questions/42633273/finding-max-of-a-column-in-vba/42633375    ws.Range("P3").Value = Application.WorksheetFunction.Min(ws.Range("k2:k" & Summary_Table_Row))          Dim GreatestDecrease As Integer     GreatestDecrease = WorksheetFunction.Match(WorksheetFunction.Min(ws.Range("k2:k" & Summary_Table_Row)), ws.Range("k2:k" & Summary_Table_Row), 0)          'print greatest decrease ticker symbol     ws.Range("O3").Value = ws.Cells(GreatestDecrease + 1, 9).Value            'find greatest voume in a column    'https://stackoverflow.com/questions/42633273/finding-max-of-a-column-in-vba/42633375    ws.Range("P4").Value = Application.WorksheetFunction.Max(ws.Range("L2:L" & Summary_Table_Row))          Dim GreatestVolume As Integer     GreatestVolume = WorksheetFunction.Match(WorksheetFunction.Max(ws.Range("L2:L" & Summary_Table_Row)), ws.Range("L2:L" & Summary_Table_Row), 0)          'print greatest increase ticker symbol     ws.Range("O4").Value = ws.Cells(GreatestVolume + 1, 9).Value          ws.Range("P2:P3").Style = "Percent"    Next wsEnd Sub
